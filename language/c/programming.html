}

void sort(int array[],int n){
    int i,j,k,t;
    for(i=0;i<n-1;i++){
        k=i;
        for(j=i+1;j<n;j++){
            if(array[j]<array[k]){
                k=j;
            }
        }
        t=array[k];
        array[k]=array[i];
        array[i]=t;
    }
}

# 求二元一次方程的解

#include<stdio.h>
#include<math.h>
int main(){
  int a,b,c;
  printf("请输入二元一次方程的三个系数");
  scanf("%d%d%d",&a,&b,&c);
  double delta,x1,x2;
  delta=b*b-4*a*c;
  if(delta>0){
  	x1=(-b+sqrt(delta))/(2*a);
  	x2=(-b-sqrt(delta))/(2*a);
	printf("x1=%f,x2=%f",x1,x2);
  }
  else if(delta==0){
  	x1=(-b)/(2*a);
  	x2=x1;
  	printf("x1=%f,x2=%f",x1,x2);
  }

  else{
  printf("无解!\n");
  }

  return 0;
}

# 求最大公约数和最小公倍数

有整数a和b

  1. a%b = c
  2. c = 0 a是最大公倍数
  3. c != 0 b=b+b 在执行1
#include <stdio.h>
int main()
{
    int m,n,gys,gbs;
    scanf("%d %d",&m,&n);
    gbs = m;
    while(gbs%n != 0) gbs += m;
    gys = m*n/gbs;
    printf("最大公约数%d,最大公倍数%d",gys,gbs);
    return 0;
}
  • 辗转相除法 有整数a和b
  1. a%b = c
  2. c = 0 b是最小公约数
  3. c != 0 则 a=b b=c 在执行1
int main()
{
    int m,n,gys,gbs,temp;
    scanf("%d %d",&m,&n);
    gbs = m; gys = n;
    while(gys != 0){
        temp = gbs%gys;
        gbs = gys;
        gys = temp;
    }
    printf("最大公约数%d,最大公倍数%d",gbs,m*n/gbs);
    return 0;
}

	for (k=2; k<=(int)sqrt((double)n); k++)
		if (n%k == 0)

}

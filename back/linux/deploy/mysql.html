# 使用Linux安装Mysql

简单介绍如何在Linux下安装Mysql

# 1、通过yum安装Mysql

下载地址：https://dev.mysql.com/downloads/repo/yum/

[root@localhost ~]# wget http://repo.mysql.com/mysql80-community-release-el7.rpm

# 2、安装

  1. 通过yum安装方法，会直接添加mysql环境变量和启动服务
[root@localhost ~]# yum -y install mysql80-community-release-el7.rpm
  1. 安装community-server
[root@localhost ~]# yum -y install mysql-community-server

# 3、启动服务并查看服务状态

systemctl start mysqld #一定要启动服务！ systemctl status mysqld #查看服务状态

[root@localhost ~]# systemctl start mysqld 
[root@localhost ~]# systemctl status mysqld

# 4、获取密码

[root@localhost ~]# grep "password" /var/log/mysqld.log

# 5、登录并求改密码

  1. 登录
[root@localhost ~]# mysql -u root -p
Enter password:
  1. 修改密码
[root@localhost ~]# mysql -u root -p
Enter password:

提示

# 6、授权

授权远程用户连接

# 选择数据库
mysql> use mysql;
# 最后授权 MySQL，允许远程用户登录访问 MySQL
mysql> update user set host = '%' where user = 'root';
# 刷新权限
mysql> flush privileges;

# 7、完成
